```python
#!/usr/bin/env python3
"""Shared helpers: run-id encode/decode and matrix expansion."""
from __future__ import annotations
import re
from itertools import product

_RUN_ID_RE = re.compile(
    r"^(?P<tech>[a-z0-9]+(?:_[a-z0-9]+)?)_(?P<coverage>\d+)x_"
    r"(?P<assembler>[a-z]+)_(?P<reference>[a-z0-9_]+)_s(?P<seed>\d+)$"
)

def make_run_id(tech, coverage, assembler, reference, seed):
    return f"{tech}_{coverage}x_{assembler}_{reference}_s{seed}"

def parse_run_id(run_id):
    m = _RUN_ID_RE.match(run_id)
    if not m:
        raise ValueError(f"bad run_id: {run_id}")
    d = m.groupdict()
    return {"tech": d["tech"], "coverage": int(d["coverage"]),
            "assembler": d["assembler"], "reference": d["reference"],
            "seed": int(d["seed"])}

def expand_matrix(cfg, default_seed):
    """Expand layers into run dicts (one per tech×assembler×coverage×ref)."""
    rows = []
    for layer in cfg["layers"].values():
        refs = layer["reference"]
        refs = [refs] if isinstance(refs, str) else refs
        for tech, ref, cov in product(cfg["assemblers"], refs, layer["coverages"]):
            if tech not in cfg["technologies"] and tech != "hybrid":
                continue
            for asm in cfg["assemblers"][tech]:
                rows.append({"tech": tech, "coverage": cov, "assembler": asm,
                             "reference": ref, "seed": default_seed})
    # multi-seed subset
    for item in cfg.get("multi_seed_subset", []):
        rows.append({**item, "seed": default_seed})
    return rows
```

```python
#!/usr/bin/env python3
"""Build repeat-injected and plasmid-bearing reference variants."""
from __future__ import annotations
import argparse, random
from pathlib import Path
from scripts.fasta_utils import read_fasta, write_fasta  # 复用现有 IO

def inject_tandem_repeat(seq, unit_len, copies, at):
    unit = seq[at:at + unit_len]
    block = unit * copies
    return seq[:at] + block + seq[at:]

def append_plasmid(records, plasmid_len, seed):
    rng = random.Random(seed)
    plasmid = "".join(rng.choice("ACGT") for _ in range(plasmid_len))
    out = dict(records)
    out["plasmid"] = plasmid
    return out

def main():
    p = argparse.ArgumentParser(description=__doc__)
    p.add_argument("--reference", default="data/reference/ecoli.fasta")
    p.add_argument("--out-dir", default="data/reference")
    p.add_argument("--seed", type=int, default=13)
    args = p.parse_args()
    records = read_fasta(args.reference)        # {name: seq}
    name, seq = next(iter(records.items()))
    # rep_v1: 少量长重复；rep_v2: 多个中等散在重复
    v1 = inject_tandem_repeat(seq, unit_len=5000, copies=3, at=1_000_000)
    rng = random.Random(args.seed)
    v2 = seq
    for _ in range(6):
        at = rng.randrange(0, len(v2) - 2000)
        v2 = inject_tandem_repeat(v2, unit_len=1500, copies=2, at=at)
    write_fasta({name: v1}, Path(args.out_dir, "ecoli_rep_v1.fasta"))
    write_fasta({name: v2}, Path(args.out_dir, "ecoli_rep_v2.fasta"))
    write_fasta(append_plasmid(records, 50_000, args.seed),
                Path(args.out_dir, "ecoli_plasmid.fasta"))

if __name__ == "__main__":
    main()
```

```python
#!/usr/bin/env python3
"""Drive ART (Illumina) and Badread (long reads) per matrix.yaml."""
from __future__ import annotations
import argparse, subprocess, yaml
from pathlib import Path

def simulate_one(tech, cfg_tech, ref_path, cov, out_prefix, seed):
    if cfg_tech["kind"] == "short":
        subprocess.run(["art_illumina", "-ss", "HS25", "-i", ref_path, "-p",
                        "-l", "150", "-f", str(cov), "-m", "350", "-s", "50",
                        "-rs", str(seed), "-o", out_prefix], check=True)
    else:
        cmd = ["badread", "simulate", "--reference", ref_path,
               "--quantity", f"{cov}x", "--seed", str(seed)]
        if cfg_tech["generation"] == "hifi":
            cmd += ["--length", "15000,3000", "--identity", cfg_tech["identity"],
                    "--error_model", "random", "--qscore_model", "random"]
        else:
            cmd += ["--length", "10000,5000", "--identity", cfg_tech["identity"]]
        with open(f"{out_prefix}.fastq", "w") as fh:
            subprocess.run(cmd, stdout=fh, check=True)

def main():
    p = argparse.ArgumentParser(description=__doc__)
    p.add_argument("--config", default="config/matrix.yaml")
    p.add_argument("--only-tech", help="只跑某 tech，便于并行/调试")
    args = p.parse_args()
    cfg = yaml.safe_load(Path(args.config).read_text())
    seed = yaml.safe_load(Path("config/seeds.yaml").read_text())["default_seed"]
    Path("data/reads").mkdir(parents=True, exist_ok=True)
    for tech, tcfg in cfg["technologies"].items():
        if args.only_tech and tech != args.only_tech:
            continue
        for cov in cfg["coverages"]:
            out = f"data/reads/{tech}_{cov}x"
            simulate_one(tech, tcfg, cfg["reference"]["orig"], cov, out, seed)

if __name__ == "__main__":
    main()
```

```python
#!/usr/bin/env python3
"""Dispatch the right assembler per technology and log RAM/time."""
from __future__ import annotations
import argparse, csv, re, subprocess, shutil
from pathlib import Path
from scripts.benchmark_common import make_run_id

GENOME = "4.6m"

def _assembler_cmd(tech, asm, reads, outdir, threads):
    r1 = f"{reads}_1.fastq"; r2 = f"{reads}_2.fastq"; long_r = f"{reads}.fastq"
    if asm == "spades":
        return ["spades.py", "-1", r1, "-2", r2, "-o", outdir, "-t", str(threads), "-m", "8"]
    if asm == "flye":
        flag = "--pacbio-hifi" if tech == "hifi" else "--nano-raw"
        return ["flye", flag, long_r, "--out-dir", outdir, "--threads", str(threads), "--genome-size", GENOME]
    if asm == "raven":
        return ["raven", "-t", str(threads), long_r]      # stdout→fasta
    if asm == "miniasm":
        return ["bash", "-c", f"minimap2 -x ava-ont {long_r} {long_r} | gzip > {outdir}/ovl.paf.gz && "
                              f"miniasm -f {long_r} {outdir}/ovl.paf.gz > {outdir}/asm.gfa"]
    if asm == "hifiasm":
        return ["hifiasm", "-o", f"{outdir}/asm", "-t", str(threads), long_r]
    if asm == "unicycler":
        return ["unicycler", "-1", r1, "-2", r2, "-l", long_r, "-o", outdir, "-t", str(threads)]
    raise ValueError(asm)

def run_one(tech, cov, asm, ref, seed, threads, log_path):
    run_id = make_run_id(tech, cov, asm, ref, seed)
    outdir = Path("assemblies", run_id); outdir.mkdir(parents=True, exist_ok=True)
    reads = f"data/reads/{tech}_{cov}x"
    cmd = _assembler_cmd(tech, asm, reads, str(outdir), threads)
    timed = ["/usr/bin/time", "-v", *cmd]
    proc = subprocess.run(timed, capture_output=True, text=True)
    Path(outdir, "stderr.txt").write_text(proc.stderr)
    # 归一化各组装器的输出到 assembly.fasta（raven/miniasm 需转换，见 stderr/各工具产物）
    _normalize_output(asm, outdir, proc.stdout)
    ram = _grep_kb(proc.stderr, r"Maximum resident set size \(kbytes\): (\d+)")
    sec = _grep_float(proc.stderr, r"Elapsed .*: ([\d:.]+)")
    _append_log(log_path, run_id, tech, cov, asm, ref, seed, ram, sec)
    return run_id
# _normalize_output/_grep_kb/_grep_float/_append_log: 见下方 Step 2 一并实现
```

```python
#!/usr/bin/env python3
"""Analysis: knee detection, variance attribution, Pareto front."""
from __future__ import annotations
import pandas as pd

def min_viable_coverage(df, gf_threshold, max_contigs):
    ok = df[(df["genome_fraction_pct"] >= gf_threshold) & (df["contigs"] <= max_contigs)]
    return None if ok.empty else int(ok["coverage"].min())

def variance_attribution(df, value="n50", factors=("tech","assembler")):
    """每个因子的组间方差占比（粗略 ANOVA 思路）。"""
    grand = df[value].mean()
    total_ss = ((df[value] - grand) ** 2).sum()
    out = {}
    for f in factors:
        ss = df.groupby(f)[value].apply(lambda g: len(g)*(g.mean()-grand)**2).sum()
        out[f] = float(ss / total_ss) if total_ss else 0.0
    return out

def pareto_front(df, cost, quality, label):
    keep = []
    for _, r in df.iterrows():
        dominated = ((df[cost] <= r[cost]) & (df[quality] >= r[quality]) &
                     ((df[cost] < r[cost]) | (df[quality] > r[quality]))).any()
        if not dominated:
            keep.append(r[label])
    return keep
```
